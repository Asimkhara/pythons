import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# --- Sample Data Generation (Replace with your actual data loading) ---
def generate_health_data(num_rows=1000):
    np.random.seed(42)  # For reproducibility
    data = {
        'age': np.random.randint(18, 80, num_rows),
        'gender': np.random.choice(['Male', 'Female'], num_rows),
        'bmi': np.random.uniform(18, 40, num_rows),
        'blood_pressure': np.random.randint(90, 160, num_rows),
        'cholesterol': np.random.randint(150, 300, num_rows),
        'smoker': np.random.choice([True, False], num_rows, p=[0.2, 0.8]), # 20% smokers
        'diabetes': np.random.choice([True, False], num_rows, p=[0.1, 0.9]), # 10% diabetes
        'heart_disease': np.random.choice([True, False], num_rows, p=[0.05, 0.95]), # 5% heart disease
        'exercise_minutes_per_week': np.random.randint(0, 300, num_rows),
        'sleep_hours': np.random.uniform(4, 9, num_rows)
    }
    return pd.DataFrame(data)

health_df = generate_health_data()

# Example: BMI categories
def bmi_category(bmi):
    if bmi < 18.5:
        return 'Underweight'
    elif 18.5 <= bmi < 25:
        return 'Normal'
    elif 25 <= bmi < 30:
        return 'Overweight'
    else:
        return 'Obese'

health_df['bmi_category'] = health_df['bmi'].apply(bmi_category)

plt.figure(figsize=(8, 6))
sns.countplot(x='bmi_category', data=health_df, order=['Underweight', 'Normal', 'Overweight', 'Obese'])
plt.title('BMI Categories')
plt.show()

# Example : Average blood pressure per age group.
health_df['age_group'] = pd.cut(health_df['age'], bins=[18, 30, 45, 60, 80], labels=['18-30', '31-45', '46-60', '61-80'])

avg_bp_age = health_df.groupby('age_group')['blood_pressure'].mean()
print('\nAverage Blood Pressure per Age Group:\n', avg_bp_age)

avg_bp_age.plot(kind='bar')
plt.title('Average Blood Pressure by Age Group')
plt.xlabel('Age Group')
plt.ylabel('Average Blood Pressure')
